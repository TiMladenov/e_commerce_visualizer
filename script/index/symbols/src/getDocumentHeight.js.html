1 /**
  2  * @author  Tihomir Mladenov, 01.08.2019
  3  * @version 1.0
  4  *
  5  * Gets the height of the client's browser window
  6  * 
  7  * @return {int} window_max_h
  8  */
  9 function getDocumentHeight() {
 10     var doc = document;
 11     window_max_h = Math.max(
 12         doc.body.scrollHeight, doc.documentElement.scrollHeight,
 13         doc.body.offsetHeight, doc.documentElement.offsetHeight,
 14         doc.body.clientHeight, doc.documentElement.clientHeight
 15     )
 16     return window_max_h;
 17 }
