  5  * Compares current scroll height to window's max height. If the end of window scroll is met it makes a call 
  6  * to a php script that fetches data from the DB to load 30 more rows, if available,
  7  * and returns them to the front end in JSON format.
  8  * Calls @see sendData for new data to be fetched.
  9  *
 10  * @param {object}  last_loaded     Gets the last added element in the span of added elements.
 11  * @param {int}     ind             Gets the index of the last added element, which is stored in the
 12  *                                  'name' html attribute.
 13  */
 14 $(document).ready (
 15     function() {
 16         $(window).scroll(function() {
 17             if($(window).scrollTop() + $(window).height() == getDocumentHeight()) {
 18                 
 19                 var last_loaded = document.getElementById("span").lastChild;
 20                 var ind = last_loaded.getAttribute("name");
 21                 sendData(0, parseInt(ind), "add");
 22             }
 23         });
 24     }
 25 );

 34 /**
 35  * @author  Tihomir Mladenov, 01.08.2019
 36  * @version 1.0
 37  * 
 38  * Sends the data and criteria entered in the front-end form to the back-end for processing.
 39  * If there's data returned, populates the UI, if not, displays an error message. Calls:
 40  * @see parseJson       to parse the returned JSON formatted string.
 41  * @see sortObjects     to sort returned rows in ascending order by product_id.
 42  * @see setCategories   to load the returned categories in the dropdown used for fitlering the rows by category.
 43  *
 44  * @param {boolean}  _new_search    If true - initiates new search, if false - just adds rows to existing ones.
 45  * @param {int}      min_index      The minimum index from which the search should be initiated.
 46  * @param {string}   _flag          A flag to point the function to the needed action: Add (to existing rows) or New (only add new rows).
 47  * 
 48  * @param {string}   searchField    Stores the text entered by the user in the search field.
 49  * @param {string}   searchFilter   Stores the selected filter option by the user - (name or description search).
 50  * 
 51  * @param {object}  http_req        Instance of createXMLHttpInstance to send an async request to the back-end.
 52  * @param {object}  form            Instance of FormData to store all the data that is to be sent to the back-end
 53  *                                  through the 'http_req'.
 54  * 
 55  * @param {string}  resp            The JSON formatted string returned by the back-end.
 56  * @param {JSON}    json            JSON object from the data in 'resp' varialbe.
 57  * @param {boolean} status_code     The status code returned from the server. Only processed if true, everything else is considered as false (problem).
 58  * @param {JSON}    data            JSON object with the actual row data from @see json.
 59  * 
 60  * @param {object}  span            References the span DOM element where all the returned rows will loaded to the front-end.
 61  * 
 62  */
 63 function sendData(_new_search, min_index, _flag) {
 64     var new_search = _new_search;
 65     var flag = _flag;
 66     var searchField = document.forms["submitForm"].elements['search_field'].value;
 67     var searchFilter = document.forms["submitForm"].elements['search_filter'].value;
 68 
 69     // var searchCategory = document.getElementById("category_filter");
 70     // var searchPromo = document.getElementById("category_promotions");
 71     // var priceFilter = document.getElementById("category_price_filter");
 72 
 73     // If text was entered by the user in the search field
 74     if(searchField != null) {
 75         var http_req = new createXMLHttpInstance();
 76         var form = new FormData(document.getElementById("submitForm"));
 77 
 78         http_req.onreadystatechange = function(){
 79             if(this.readyState==4 && this.status==200) {
 80                 var resp = this.responseText;
 81                 var json = parseJson(resp);
 82                 
 83                 var status_code = json.status_code;
 84                 var data = json.data;
 85                 data = sortObjects(data);
 86 
 87                 var span = document.getElementById("span");
 88                 if(new_search == 1) {
 89                     span.innerHTML = "";
 90                 }
 91                 
 92                 var search_button = document.getElementById("search_button");
 93                 
 94                 // Loads the data from the JSON into the front-end, if there was actual data returned.
 95                 if(status_code == true) {
 96                     if(data.length > 0) {
 97                         var categories = new Array(data.length);
 98                         for(var i = 0; i < data.length; i++) {
 99                             var my_module = "\
100                             <article data-is-promo="+data[i].product_is_promo+" data-product-category="+data[i].product_category+" \
101                             data-product-price="+data[i].product_price+" name="+data[i].product_id+" class=\"row mt-2 border rounded\">\
102                                 <div class=\"item col-2 text-left border-right align-items-center\">"+data[i].product_name+"</div>\
103                                 <div class=\"d-flex item col-2 text-left border-right align-items-center\">\
104                                 <img src="+data[i].product_image+" class=\"img-thumbnail\" alt=\"Product image\" onclick=\"window.open(this.src)\">\
105                                 </div>\
106                                 <div class=\"d-flex item col-3 text-left border-right align-items-center\">"+data[i].product_price+', '+data[i].price_curr +"</div>\
107                                 <div class=\"d-flex item description col-3 text-justify border-right\">"+data[i].product_descr+"</div>\
108                                 <div class=\"d-flex item col-2 text-left border-left align-items-center justify-content-center\">\
109                                 <a href="+data[i].product_shop_url+" class=\"visit_vendor\" title=\"Visit vendor's website\" onclick=\"countClick(this)\">Vendor</a>\
110                                 </div>\
111                             </article>"
112                             span.innerHTML += my_module;
113                             categories.push(data[i].product_category);
114                             search_button.setAttribute("onClick", "(sendData(1,"+data[i].product_id+", 'new'))");
115                             visit_links = document.getElementsByClassName("visit_vendor");
116                             for (var visit_ind = 0; visit_ind < visit_links.length; visit_ind++) {
117                                 visit_links[i].setAttribute("onclick", "countClick(this)");
118                             }
119                         }
120                         resetFilters();
121                         setCategories(categories);
122                     }
123                 } else if(status_code == false) {
124                     span.innerHTML = "<p>Ooops, we couldn't find anything related to your search.</p>";
125                 }
126             }
127         }
128         http_req.open("POST", "main.php", true);
129         form.append('search_field', searchField);
130         form.append('search_filter', searchFilter);
131         if((min_index != 1 && new_search == 1) || min_index == null) {
132             min_index = 1;    
133         }
134         if(flag == null || flag == "") {
135             flag == "new";
136         }
137         form.append('min_index', min_index);
138         form.append('flag', flag);
139         http_req.send(form);
140     } else {
141         alert("Please provide a search criteria.");
142     }
143 }

146 /**
147  * @author  Tihomir Mladenov, 01.08.2019
148  * @version 1.0
149  * 
150  * Before loading the returned json objects into the FE, this function is called to compare their ID's and sorts
151  * them in an ascending order with Bubble sort.
152  *
153  * @param   {array} data    json object with the returned rows by the back-end
154  * @return {array}
155  */
156 function sortObjects(data) {
157     var arr = data;
158     for(var i = 0; i < arr.length; i++) {
159         for(var j = i + 1; j < arr.length; j++) {
160             if(arr[i].product_id > arr[j].product_id) {
161                 var tmp_obj = arr[i];
162                 arr[i] = arr[j];
163                 arr[j] = tmp_obj;
164             }
165         }
166     }
167     return arr;
168 }

170 // Sets the categories in the categories dropdown, which is used by the user
171 // to filter through the content. Filters the duplicates
172 function setCategories(categories) {
173     var category_filter = document.getElementById("category_filter");
174     category_filter.innerHTML = "<option selected value=\"0\"></option>";
175     var tmp = "";
176     categories = categories.sort();
177     for (var i = 0; i < categories.length - 1; i++) {
178         var j = i + 1;
179         if(categories[i] !== categories[j]){
180             if(categories[i] != null) {
181                 tmp = "<option value="+categories[i]+">"+categories[i]+"</option>";
182                 category_filter.innerHTML += tmp;
183             }
184         }
185     }
186 }

188 // Creates an HttpInstance for async call to the BE.
189 function createXMLHttpInstance() {
190     var request = null;
191     if (window.XMLHttpRequest) {
192         request = new XMLHttpRequest();
193     } else if (window.ActiveXObject) {
194         try {
195             request = new ActiveXObject("Msxml2.XMLHTTP");
196         } catch (e) {
197             try {
198                 request = new ActiveXObject("Microsoft.XMLHTTP");
199             } catch (e) {
200                 alert("XHR not created");
201             }
202         }
203     }
204     return request;
205 }

207 // Parses the returned json formatted string to json object
208 function parseJson(text) {
209     try {
210         var json = JSON.parse(text);
211     } catch(e) {
212         return false;
213     }
214     return json;
