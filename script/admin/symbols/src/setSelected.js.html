1 /**
  2  * @author Tihomir Mladenov, 29.07.2019
  3  * @version 1.0
  4  * 
  5  * Displays the selected category from the category dropdown.
  6  * 
  7  * @param {object} selectedField    References the DOM element where the selected
  8  *                                  category and its properties will be displayed.
  9  * @param {object} selected         References the selected element from the dropdown
 10  *                                  list with all categories and properties.
 11  * @param {String} text             Stores the text from the selected category in the
 12  *                                  dropdown.
 13  *
 14  */
 15 function setSelected() {
 16     var selectedField = document.getElementById("category_selected");
 17     selectedField.innerHTML = "";
 18     
 19     var selected = document.getElementById("category_edit_id");
 20     var text = selected.options[selected.selectedIndex].value + " " + selected.options[selected.selectedIndex].text;
 21     selectedField.innerHTML = text;
 22 }
