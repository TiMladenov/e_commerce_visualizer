1 /**
  2  * @author  Tihomir Mladenov, 29.07.2019
  3  * @version 1.0
  4  * 
  5  * Called when the user clicks the checkbox to hide the entries with 0 visits.
  6  * 
  7  * @param {object} container    References the tr container the holds td's
  8  * @param {int}    visits       Stores the visit count for particular tr
  9  *
 10  * @param {object} checkbox
 11  */
 12 function filter(checkbox) {
 13     //The checkbox is checked
 14     if(checkbox.checked) {
 15         var container = document.getElementById("table_price_range_body").childNodes;
 16         for(var i = 0; i < container.length; i++) {
 17             var visits = parseInt(container[i].childNodes[6].innerHTML);
 18             //Hides the table row if the price in it is 0
 19             if(visits <= 0) {
 20                 container[i].setAttribute("hidden", true);
 21             }
 22         }
 23     // The checkbox is unchecked
 24     } else {
 25         //Unhides all table rows
 26         var container = document.getElementById("table_price_range_body").childNodes;
 27         for(var i = 0; i < container.length; i++) {
 28             container[i].removeAttribute("hidden");
 29         }
 30     }
 31 }
