1 /**
  2  * @author  Tihomir Mladenov, 29.07.2019
  3  * @version 1.0
  4  * 
  5  * Receives the new category name for a product as provided by the user.
  6  * If the string is empty, a warning is triggered, else also gets the 
  7  * ID of the item and sends the text and id to the BE for processing.
  8  * Returns "true" or "false" depending on successful update or not.
  9  * 
 10  * @param {object}      selectedField   References the field with the 
 11  *                                      selected category.
 12  * @param {object}      selected        References the selected dropdown
 13  *                                      element.
 14  * @param {string}      text            Contains the text of the selected
 15  *                                      dropdown element.
 16  * @param {string}      id              Contains the id of the edited element
 17  * @param {object}      new_category    The input for the new category text.
 18  * @param {string}      obj_loc         The URL to the BE script.
 19  * 
 20  * @return {boolean}
 21  */
 22 function updateCat() {
 23     var selectedField = document.getElementById("category_selected")
 24     if(selectedField.innerHTML == "" || selectedField.innerHTML == null) {
 25         alert("Please select a category that you want to update first!");
 26         return false;
 27     } else {
 28         var selected = document.getElementById("category_edit_id");
 29         var text = selected.options[selected.selectedIndex];
 30         var id = text.getAttribute("value");
 31         var new_category = document.getElementById("category_new").value;
 32         if (new_category == "" || new_category == null) {
 33             alert("Please provide a valid category name");
 34         } else {
 35             var obj_loc = window.location.origin;
 36             obj_loc = obj_loc + '/'+ 'admin.php';
 37             
 38             $.ajax({
 39             type : 'POST',
 40             url : obj_loc,
 41             data : {
 42                 'control' : 'update',
 43                 'product_id' : id,
 44                 'category' : new_category
 45             },
 46             async : false,
 47             success : function(result) {
 48                 if(result == true) {
 49                     alert("The category was successfully updated.");
 50                     window.location.reload(true);
 51                 } else {
 52                     alert("Ooops, the category could not be updated...");
 53                 }
 54             }
 55         });
 56     }
 57 }
 58 }
