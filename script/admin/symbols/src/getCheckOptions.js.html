1 /**
  2  * @author  Tihomir Mladenov, 29.07.2019
  3  * @version 1.0
  4  * 
  5  * Gets the selected values from the selected checkboxes and creates two
  6  * arrays with parameters, which are then sent to the BE to make a DB
  7  * query. The results are then returned in json format and passed to
  8  * @see parseResponse()
  9  * 
 10  * @param {object}  chk1_parent     References the parent DOM object of all
 11  *                                  checkboxes.
 12  * @param {array}   qry_arr         Stores the query parameters for the first
 13  *                                  three checkboxes.
 14  * @param {array}   qry_arr_second  Stores the query parameters from the fourth
 15  *                                  checkbox.
 16  * @param {string}  obj_loc         URL to the script to handle the request.
 17  *
 18  * @return {JSON}
 19  */
 20 function getCheckOptions() {
 21     var chk1_parent = document.getElementById("chk_parent").childNodes;
 22     var qry_arr = [];
 23     var qry_arr_second = [];
 24     for(var i = 0; i < chk1_parent.length; i++) {
 25         if(chk1_parent[i].nodeName != '#text') {
 26             if(chk1_parent[i].checked == true){
 27                 if(i >= 13) {
 28                     qry_arr_second.push(chk1_parent[i].getAttribute("name").split(" "));
 29                 } else {
 30                     qry_arr.push(chk1_parent[i].getAttribute("name").split(" "));
 31                 }
 32             }
 33         }
 34     }
 35     if(qry_arr.length <= 0 && qry_arr_second.length <=0) {
 36         alert("Please select search criteria!");
 37         return false;
 38     }
 39     qry_arr = qry_arr.toString();
 40     qry_arr_second = qry_arr_second.toString();
 41 
 42     var obj_loc = window.location.origin;
 43     obj_loc = obj_loc + '/'+ 'admin.php';
 44 
 45     $.ajax({
 46         type : 'POST',
 47         url : obj_loc,
 48         data : {
 49             'control' : 'browse',
 50             'query_one' : qry_arr,
 51             'query_two' : qry_arr_second
 52         },
 53         async : false,
 54         success : function(result) {
 55             parseResponse(result);
 56         }
 57     });
 58     
 59 }

 61 /**
 62  * @author  Tihomir Mladenov, 29.07.2019
 63  * @version 1.0
 64  * 
 65  * Receives a JSON formatted string which it then parses to a
 66  * JSON object. Loops through the JSON, gets the data for
 67  * each object and populates the UI with it.
 68  * 
 69  * @see getCkeckOptions()
 70  * 
 71  * @param {string}  result                  JSON formatted string with the fetched
 72  *                                          data from the BE.
 73  * @param {JSON}    json                    JSON object from the result variable.
 74  * @param {JSON}    main_content            Stores the data from the first query, if
 75  *                                          returned.
 76  * @param {json}    secondary_content       Stores the data from the second query, if
 77  *                                          returned.
 78  * @param {object}  table_visitor_data      References the table for the visitor data
 79  * @param {object}  table_shop_data         References the table for the shop data
 80  * @param {object}  table_price_range_data  References the table for the price range data
 81  * @param {object}  table_unique_visit_data References the table for the unique user visits
 82  * @param {string}  content                 Used to store each new row before it is added to the
 83  *                                          DOM parent element.
 84  */
 85 function parseResponse(result) {
 86     var json = JSON.parse(result);
 87     var main_content = json.main;
 88     var secondary_content = json.secondary;
 89 
 90     var table_visitor_data = document.getElementById("table_visitor_body");
 91     var table_shop_data = document.getElementById("table_shop_body");
 92     var table_price_range_data = document.getElementById("table_price_range_body");
 93     var table_unique_visit_data = document.getElementById("table_unique_visit_body");
 94     table_visitor_data.innerHTML = "";
 95     table_shop_data.innerHTML = "";
 96     table_price_range_data.innerHTML = "";
 97     table_unique_visit_data.innerHTML = "";
 98     
 99     var content = "";
100 
101     for(var i = 0; i < main_content.length; i++) {
102         if(main_content[i].product_name != null && main_content[i].product_category != null) {
103             if(main_content[i].product_category_visits == null) {
104                 main_content[i].product_category_visits = 0;
105             }
106             content = "<tr><th scope=\"row\">"+(i+1)+"</th><td>"+main_content[i].product_name+"</td><td>"+main_content[i].product_category+"</td><td>"+main_content[i].product_category_visits+"</td></tr>";
107             table_visitor_data.innerHTML += content;
108         }
109         if(main_content[i].shop_name != null) {
110             for(var j = i + 1; j < main_content.length; j++) {
111                 if(i == 0) {
112                     if(main_content[i].shop_visits == null) {
113                         main_content[i].shop_visits = 0;
114                     }
115                     content = "<tr><th scope=\"row\">"+j+"</th><td>"+main_content[i].shop_name+"</td><td>"+main_content[i].shop_visits+"</td></tr>";
116                     table_shop_data.innerHTML += content;
117                     break;
118                 } else {
119                     if(main_content[i].shop_name != main_content[j].shop_name) {
120                         if(main_content[i].shop_visits == null) {
121                             main_content[i].shop_visits = 0;
122                         }
123                         content = "<tr><th scope=\"row\">"+(i+1)+"</th><td>"+main_content[i].shop_name+"</td><td>"+main_content[i].shop_visits+"</td></tr>";
124                         table_shop_data.innerHTML += content;
125                     }
126                 }
127             }
128         }
129         if(main_content[i].price != null) {
130             if(main_content[i].price_range_visits == null) {
131                 main_content[i].price_range_visits = 0;
132             }
133 
134             var price = parseInt(main_content[i].price);
135             var price_range = "";
136 
137             var range_visits = 0;
138             var element = null;
139             // Checks the price value. Adds it to the respectable price range
140             if(price >= 0 && price < 5) {
141                 price_range = "0 - 5";
142             } else if(price >= 5 && price < 10) {
143                 price_range = "5 - 10";
144             } else if(price >= 10 && price < 20) {
145                 price_range = "10 - 20";
146             } else if(price >= 20 && price < 40) {
147                 price_range = "20 - 40";
148             } else if(price >= 40 && price < 80) {
149                 price_range = "40 - 80";
150             } else if(price >= 80 && price < 160) {
151                 price_range = "80 - 160";
152             } else if(price >= 160) {
153                 price_range = "160 - ...";
154             }
155             content = "<tr><th scope=\"row\">"+(i+1)+"</th><td>"+main_content[i].product_name+"</td><td>"+main_content[i].price+"</td><td>"+main_content[i].price_curr+"</td>\
156             <td>"+price_range+"</td><td>"+main_content[i].price_range_visits+"</td></tr>";
157             table_price_range_data.innerHTML += content;
158 
159             if(main_content[i].price_range_visits > 0) {
160                 setPriceRangeVisitsTotal(price_range, main_content[i].price_range_visits);
161             }
162         }
163     }
164     if (secondary_content != "" && secondary_content != null) {
165         for(var i = 0; i < secondary_content.length; i++) {
166             content = "<tr><th scope=\"row\">"+(i+1)+"</th><td>"+secondary_content[i].product_id+"</td><td>"+secondary_content[i].product_name+"</td><td>"+secondary_content[i].product_visits+"</td></tr>"
167             table_unique_visit_data.innerHTML += content;
168         }
169     }
170 }

172 // Sets the total visit count for each price-range.
173 function setPriceRangeVisitsTotal(price_range, content_visits) {
174     element = document.getElementById(price_range);
175     range_visits = parseInt(element.innerText);
176     range_visits = range_visits + content_visits;
177     element.innerText = range_visits;
178 }
